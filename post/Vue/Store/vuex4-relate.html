  state: {
    userData: {},

    updateUserData(state, newUserData) {
      state.userData = newUserData;
    },
  },
  actions: {
    async getUserData(store) {
      await useApi(store, 1000);
      store.commit('updateUserData', userData);
    },
  },
});

  el: '#app',
  store,
  created() {
    this.$store.dispatch('getUserData');
