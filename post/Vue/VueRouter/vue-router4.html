# 與composition組合

import { watch } from 'vue';
import { useRoute, useRouter } from 'vue-router';
const router = useRouter() //事件中獲取 router，進行路由跳轉等等。
const route = useRoute() //獲取路由後再進行動作，比如說監聽路由等等

const initPage = () => {
  router.push({
    name: 'Index'
  })
}

const showParams = () => {
  console.log(route.path)
}

watch(route, () => {
  showParams()
})
