//物件取key & value值
var obj = {
name: 'maju',
age: 10
};

console.log(Object.keys(obj)); 把要轉換的對象丟入
//得到["name", "age"]
console.log(Object.values(obj)); 
//得到["maju", 10]

要調用時要再拿東西接 => const objKeys = Object.keys(obj);


//構造函數

/*
1.構造函數就是函數，只是有自己的屬性和方法
2.為了與普通函數區別，構造函數第一個字通常為大寫
* 構造函數特點 *
-函數內部使用'this',代表所要生成的對象實例
-生成對象的時候, 必須使用'new'命令

*/

let Maju = function(){
this.size = 'big';
}

let Q = new Maju();
console.log(Q.size) //'big'
